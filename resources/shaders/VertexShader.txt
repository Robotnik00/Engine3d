#version 130

in vec3 VertexPosition;
in vec3 normals;
in vec3 tex;

out vec3 v;
out vec4 N;
out vec3 texcoords;

void main()
{
	mat4 transform = gl_ModelViewMatrix;
	transform[3][0] = 0;
	transform[3][1] = 0;
	transform[3][2] = 0;
	
	N = normalize(transform * vec4(normals,1.0));
	v = VertexPosition;
	texcoords = tex;

	gl_Position = gl_ModelViewProjectionMatrix * vec4(VertexPosition,1.0);
}

#version 130

in vec3 v;    
in vec4 N;
in vec3 texcoords;
in vec4 lightf;

out vec4 FragColor;

uniform sampler2D Tex1;

uniform vec4 light;
uniform vec4 color;

void main (void)  
{  
   float dotprod = dot(lightf, N);
	
   vec4 texcolor = texture(Tex1,texcoords.xy);
	
   FragColor = (texcolor + color) * dotprod; 
}
